Coding the Logic


// 1. Define the Callback (What to do later)
function showProduct(productName) {
    console.log("Rendering UI for: " + productName);
}

// 2. Define the Main Task (Server Request)
function fetchFromServer(productId, myCallback) {
    console.log("Fetching data...");
    // Simulating delay
    setTimeout(() => {
        let data = "Nike Shoes"; // Data arrived!
        myCallback(data);      // Executing callback
    }, 2000);
}

// 3. Trigger It
fetchFromServer(101, showProduct);
